#include <iostream>
using namespace std;

int n, m;
int ch[10], res[10];

void DFS(int L, int x) {
	if (L == m) {
		for (int i = 0; i < m; i++) {
			cout << res[i] << " ";
		}
		cout << "\n";
		return;
	}

	for (int i = x; i <= n; i++) {
		if (ch[i] == 0) {
			ch[i] = 1;
			res[L] = i;
			DFS(L + 1, i + 1);
			ch[i] = 0;
		}
	}
}

int main() {
	cin >> n >> m;
	DFS(0, 1);
}
