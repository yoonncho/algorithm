#include <iostream>
using namespace std;

int n, m;
int res[10];

void DFS(int L, int x) {
	if (L == m) {
		for (int i = 0; i < m; i++) {
			cout << res[i] << " ";
		}
		puts("");
	}
	else {
		for (int i = x; i <= n; i++) {
			res[L] = i;
			DFS(L + 1, i);
		}
	}
}

int main() {
	cin >> n >> m;
	DFS(0, 1);
}
