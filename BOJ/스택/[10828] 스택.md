			}

			}
		}

		if (str == "size") {
			cout << st.size() << endl;
		}

		if (str == "empty") {
			if (st.empty()) cout << "1" << endl;
			else cout << "0" << endl;
		}

		if (str == "top") {
			if (!st.empty()) cout << st.top() << endl;
			else cout << "-1" << endl;
		}
	}
	return 0;
}

#include <iostream>
using namespace std;

int s, st[10001];

void push(int x) {
	st[s++] = x;
}

int pop() {
	if (s == 0) return -1;
	else return st[--s];
}

int size() {
	return s;
}

int isEmpty() {
	if (s == 0) return 1;
	else return 0;
}

int top() {
	if (s == 0) return -1;
	return st[s-1];
}


int main() {
	int n;
	string str;
	stack<int> st;

		cin >> str;
