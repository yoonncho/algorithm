#include <iostream>
#include <string>
#include <stack>
using namespace std;

int main() {
	int n;
	string str;

	cin >> n;

	for (int i = 0; i < n; i++) {
		cin >> str;

		stack<char> st;
		string res = "YES";

		for (int j = 0; j < str.length(); j++) {
			char par = str[j];
			if (par == '(') st.push(par);
			else {
				if (st.empty()) res = "NO";
				else st.pop();
			}
		}

		if (!st.empty()) res = "NO";
		cout << res << endl;
	}
	return 0;
}
