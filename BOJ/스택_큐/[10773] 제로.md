#include <iostream>
#include <stack>
using namespace std;

int main() {
	int n, num, sum = 0;
	stack<int> st;
	cin >> n;
	for (int i = 0; i < n; i++) {
		cin >> num;
		if (num != 0) st.push(num);
		else st.pop();
	}

	while (!st.empty()) {
		sum += st.top();
		st.pop();
	}

	cout << sum;
	return 0;
}
