#include <stdio.h>
#include <algorithm>
#include <vector>
using namespace std;

int n, m;
vector<int> arr;
vector<int> num;

int Find(int target) {
	int mid, lt = 0, rt = n-1;
	while (lt <= rt) {
		mid = (lt + rt) / 2;
		if (arr[mid] == target) return 1;
		else if (arr[mid] < target) lt = mid + 1;
		else rt = mid - 1;
	}
	return 0;
}

int main() {
	int tmp;
	scanf("%d", &n);
	for (int i = 0; i < n; i++) {
		scanf("%d", &tmp);
		arr.push_back(tmp);
	}
	sort(arr.begin(), arr.end());

	scanf("%d", &m);
	for (int i = 0; i < m; i++) {
		scanf("%d", &tmp);
		num.push_back(tmp);
	}

	for (int i = 0; i < m; i++) {
		printf("%d\n", Find(num[i]));
	}
	return 0;
}
