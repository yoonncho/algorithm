#include <stdio.h>
#include <algorithm>
#include <vector>
using namespace std;

int n, m, res, sum;
vector<int> v;
int ch[101];


void DFS(int L) {
  if (L == 3) {
    if (sum <= m) res = max(res, sum);
    return;
  }
  else {
    for (int i = 0; i < n; i++) {
      if (ch[i] == 0) {
        ch[i] = 1;
        sum += v[i];
        DFS(L + 1);
        sum -= v[i];
        ch[i] = 0;
      }
    }
  }
}

int main() {
  scanf("%d %d", &n, &m);
  int tmp;
  for (int i = 0; i < n; i++) {
    scanf("%d", &tmp);
    v.push_back(tmp);
  }

  DFS(0);
  printf("%d\n", res);
  return 0;
}
